  {{- range $i, $e := until (int .Values.broker.replicaCount) }}

from Metadata and use the Kafka protocol's standard retry behaviour for leader

  {{- range $i, $e := until (int .Values.broker.replicaCount) }}

  {{- end }}

listeners.external.hostnamePattern with your DNS pattern.
